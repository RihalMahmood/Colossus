Backend: 
    1. npm init -y
    2. npm install express@4.18.2
    3. npm install mongoose
    4. npm i cors
    5. npm install bcryptjs jsonwebtoken dotenv     //For login auth
    6. npm uninstall bcryptjs && npm install bcryptjs@2.4.3     //If 'bcryptjs' version is 3 or above
    7. npm install -D nodemon       //Auto restarts server when a file is edited

Frontend:
    1. npm create vite@latest
    2. npm i react-router && npm install react-router-dom
    3. npm i react-hot-toast
    4. npm install -D tailwindcss@3 postcss autoprefixer && npx tailwindcss init -p
    5. npm i daisyui -D
    6. npm i lucide-react
    7. npm i axios

Git: 
    1. git rm --cached filename.ext     //To remove a file which was committed in past
